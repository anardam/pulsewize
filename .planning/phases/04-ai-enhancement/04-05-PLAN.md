**buildComparePrompt(profiles: NormalizedProfile[], platform: string): string**
```
Prompt structure:
- States: "You are a social media analyst comparing ${profiles.length} competitor profiles on ${platform}."
- Includes JSON of all profiles: profiles.map(p => JSON.stringify(p, null, 2)).join('\n---\n')
- Requests output as CompetitorComparisonReport JSON with:
  - metricsTable: array of { metric, values: [{username, value, isLeader}] } for: Followers, Following, Posts/Videos, Engagement Rate, Verification Status, Bio Link Present, Posting Frequency
  - narrative: 3-4 paragraph analysis of differences and opportunities (D-08)
  - opportunities: array of 3-5 specific differentiation opportunities
- Instructs: "isLeader: true for the profile leading in that metric. Return ONLY valid JSON."
```

**buildCalendarPrompt(profile: NormalizedProfile | ManualProfileInput, platform: string): string**
```
Prompt structure:
- States: "You are a content strategist creating a 30-day content calendar for @${profile.username} on ${platform}."
- Includes profile JSON
- Requests output as ContentCalendarReport JSON with:
  - weeks: exactly 4 CalendarWeek objects (weekNumber 1-4)
  - Each week has 7 day keys (monday through sunday)
  - Each day is a CalendarEntry with: contentIdea, optimalPostingTime (timezone-aware), contentType (reel/story/tweet/video/carousel/post), captionDraft (full caption, 100-150 chars), hashtags (5-8 tags), mediaSuggestion, engagementPrediction
  - niche: inferred from profile bio/content
- Instructs: "Base posting times on the creator's audience timezone from their location/bio if available, else default to EST. Return ONLY valid JSON."
```

```typescript
import { NextRequest, NextResponse } from "next/server";
import { createSupabaseServer } from "@/lib/supabase/server";
import { getScraper } from "@/lib/scrapers/registry";
import { buildComparePrompt } from "@/lib/prompt";
import { createOpenRouterClient } from "@/lib/ai/openrouter-client";
import { CLAUDE_SYNTHESIS_MODEL, SYNTHESIS_MAX_TOKENS } from "@/lib/ai/models";
import type { CompetitorComparisonReport, NormalizedProfile } from "@/lib/types";
import type { NormalizedProfile as ScraperNormalizedProfile } from "@/lib/scrapers/types";

export async function POST(request: NextRequest) {
  const supabase = await createSupabaseServer();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) {
    return NextResponse.json({ success: false, error: "Authentication required" }, { status: 401 });
  }

  try {
    const body = await request.json();
    const { platform, usernames } = body as { platform: string; usernames: string[] };

    if (!platform || !usernames || usernames.length < 2 || usernames.length > 3) {
      return NextResponse.json(
        { success: false, error: "Provide 2-3 usernames on the same platform (D-09)" },
        { status: 400 }
      );
    }

    // Scrape all profiles in parallel — re-scrape rather than use saved reports (research recommendation)
    const scraper = getScraper(platform);
    if (!scraper) {
      return NextResponse.json({ success: false, error: `Unsupported platform: ${platform}` }, { status: 400 });
    }

    const scrapeResults = await Promise.allSettled(
      usernames.map((u) => scraper.scrape(u))
    );

    const profiles: ScraperNormalizedProfile[] = scrapeResults
      .filter((r): r is PromiseFulfilledResult<{ success: true; profile: ScraperNormalizedProfile }> =>
        r.status === "fulfilled" && r.value.success && !!r.value.profile
      )
      .map((r) => r.value.profile as ScraperNormalizedProfile);

    if (profiles.length < 2) {
      return NextResponse.json(
        { success: false, error: "Could not fetch at least 2 profiles. Check usernames and try again." },
        { status: 422 }
      );
    }

    const prompt = buildComparePrompt(profiles as unknown as NormalizedProfile[], platform);
    const client = createOpenRouterClient();
    const completion = await client.chat.completions.create({
      model: CLAUDE_SYNTHESIS_MODEL,
      max_tokens: SYNTHESIS_MAX_TOKENS,
      messages: [{ role: "user", content: prompt }],
    });

    const text = completion.choices[0]?.message?.content ?? "";
    let jsonStr = text.trim();
    const jsonMatch = jsonStr.match(/```(?:json)?\s*([\s\S]*?)```/);
    if (jsonMatch) jsonStr = jsonMatch[1].trim();
    const firstBrace = jsonStr.indexOf("{");
    const lastBrace = jsonStr.lastIndexOf("}");
    if (firstBrace === -1 || lastBrace === -1) {
      return NextResponse.json({ success: false, error: "Failed to parse comparison report" }, { status: 500 });
    }
    jsonStr = jsonStr.slice(firstBrace, lastBrace + 1);
    const report: CompetitorComparisonReport = JSON.parse(jsonStr);

    return NextResponse.json({ success: true, report });
  } catch (error) {
    const message = error instanceof Error ? error.message : "Unknown error occurred";
    return NextResponse.json({ success: false, error: message }, { status: 500 });
  }
}
```
