```typescript
export async function runInstagramCascade(username: string): Promise<ScraperResult> {
  const scrapers: Array<() => Promise<ScraperResult>> = [...];
  for (const scrape of scrapers) {
    const result = await scrape();
    if (result.success && result.profile) return result;
  }
  return { success: false, requiresManualEntry: true, error: "All scrapers failed" };
}
```

5. Implement `scrapeYouTubeChannelPrimary(handle: string): Promise<ScraperResult>`:
   - Read `YOUTUBE_API_KEY` from `process.env.YOUTUBE_API_KEY` — if missing, return `{ success: false, error: "YOUTUBE_API_KEY not configured" }`
   - Strip leading `@` from handle: `const cleanHandle = handle.replace(/^@/, "").trim()`
   - Call `youtube.channels.list({ auth: apiKey, part: ["snippet", "statistics"], forHandle: cleanHandle })`
   - If response has no items, try `forUsername: cleanHandle` as internal fallback (legacy handles)
   - Parse the first channel item through `YouTubeChannelSchema.safeParse()` — if parse fails, return error
   - Map to `YouTubeProfile`:

     ```typescript
     const profile: YouTubeProfile = {

     ```
